57template <typename SampleType>
58void Biquads<SampleType>::setFrequency (SampleType newFreq)
59{
60 jassert(minFreq <= newFreq && newFreq <= maxFreq);
61
62 if (hz != newFreq)
63 {
64 hz = juce::jlimit(minFreq, maxFreq, newFreq);
65
66 omega = (hz * ((pi * two) / static_cast <SampleType>(sampleRate)));
67 cos = (std::cos(omega));
68 sin = (std::sin(omega));

70 calculateCoefficients();
71 }

241template <typename SampleType>
242SampleType Biquads<SampleType>::directFormI(int channel, SampleType inputSample)
243{
244 auto& Xn1 = Xn_1[(size_t)channel];
245 auto& Xn2 = Xn_2[(size_t)channel];
246 auto& Yn1 = Yn_1[(size_t)channel];
247 auto& Yn2 = Yn_2[(size_t)channel];
248
249 auto& Xn = inputSample;
250 auto& Yn = outputSample;
251
252 Yn = ((Xn * b0) + (Xn1 * b1) + (Xn2 * b2) + (Yn1 * a1) + (Yn2 * a2));
253
254 Xn2 = Xn1, Yn2 = Yn1;
255 Xn1 = Xn, Yn1 = Yn;
256
257 return Yn;
258}

268template <typename SampleType>
269SampleType Biquads<SampleType>::directFormII(int channel, SampleType inputSample)
270{
271 auto& Wn1 = Wn_1[(size_t)channel];
272 auto& Wn2 = Wn_2[(size_t)channel];
273
274 auto& Wn = loop;
275 auto& Xn = inputSample;
276 auto& Yn = outputSample;
277
278 Wn = (Xn + ((Wn1 * a1) + (Wn2 * a2)));

346template <typename SampleType>
347void Biquads<SampleType>::calculateCoefficients()
348{
349 alpha = (sin * (one - q));
350 a = (std::pow(SampleType(10), (g * SampleType(0.05))));
351 sqrtA = ((std::sqrt(a) * two) * alpha);
352
353 switch (filtType)
354 {
355 case filterType::lowPass2:
356
357 b_0 = (one - cos) / two;
358 b_1 = one - cos;
359 b_2 = (one - cos) / two;
360 a_0 = one + alpha;
361 a_1 = minusTwo * cos;
362 a_2 = one - alpha;
363
364 break;
365
366
367 case filterType::lowPass1:
368
369 b_0 = omega / (one + omega);
370 b_1 = omega / (one + omega);
371 b_2 = zero;
372 a_0 = one;
373 a_1 = minusOne * ((one - omega) / (one + omega));
374 a_2 = zero;
375
376 break;
377
378
379 case filterType::highPass2:
380
381 b_0 = (one + cos) / two;
382 b_1 = minusOne * (one + cos);
383 b_2 = (one + cos) / two;
384 a_0 = one + alpha;
385 a_1 = minusTwo * cos;
386 a_2 = one - alpha;
387
388 break;
389
390
391 case filterType::highPass1:
392
393 b_0 = one / (one + omega);
394 b_1 = (one / (one + omega)) * minusOne;
395 b_2 = zero;
396 a_0 = one;
397 a_1 = ((one - omega) / (one + omega)) * minusOne;
398 a_2 = zero;
399
400 break;
401
402
403 case filterType::bandPass:
404
405 b_0 = sin / two;
406 b_1 = zero;
407 b_2 = minusOne * (sin / two);
408 a_0 = one + alpha;
409 a_1 = minusTwo * cos;
410 a_2 = one - alpha;
411
412 break;
413
414
415 case filterType::bandPassQ:
416
417 b_0 = alpha;
418 b_1 = zero;
419 b_2 = minusOne * alpha;
420 a_0 = one + alpha;
421 a_1 = minusTwo * cos;
422 a_2 = one - alpha;
423
424 break;
425
426
427 case filterType::lowShelf2:
428
429 b_0 = (((a + one) - ((a - one) * cos)) + sqrtA) * a;
430 b_1 = (((a - one) - ((a + one) * cos)) * two) * a;
431 b_2 = (((a + one) - ((a - one) * cos)) - sqrtA) * a;
432 a_0 = ((a + one) + ((a - one) * cos)) + sqrtA;
433 a_1 = ((a - one) + ((a + one) * cos)) * minusTwo;
434 a_2 = ((a + one) + ((a - one) * cos)) - sqrtA;
435
436 break;
437
438
439 case filterType::lowShelf1:
440
441 b_0 = one + ((omega / (one + omega)) * (minusOne + (a * a)));
442 b_1 = (((omega / (one + omega)) * (minusOne + (a * a))) - ((one - omega) / (one + omega)));
443 b_2 = zero;
444 a_0 = one;
445 a_1 = minusOne * ((one - omega) / (one + omega));
446 a_2 = zero;
447
448 break;
449
450
451 case filterType::lowShelf1C:
452
453 b_0 = one + ((omega / a) / (one + (omega / a)) * (minusOne + (a * a)));
454 b_1 = ((((omega / a) / (one + (omega / a))) * (minusOne + (a * a))) - ((one - (omega / a)) / (one + (omega / a))));
455 b_2 = zero;
456 a_0 = one;
457 a_1 = minusOne * ((one - (omega / a)) / (one + (omega / a)));
458 a_2 = zero;
459
460 break;
461
462
463 case filterType::highShelf2:
464
465 b_0 = (((a + one) + ((a - one) * cos)) + sqrtA) * a;
466 b_1 = (((a - one) + ((a + one) * cos)) * minusTwo) * a;
467 b_2 = (((a + one) + ((a - one) * cos)) - sqrtA) * a;
468 a_0 = ((a + one) - ((a - one) * cos)) + sqrtA;
469 a_1 = ((a - one) - ((a + one) * cos)) * two;
470 a_2 = ((a + one) - ((a - one) * cos)) - sqrtA;
471
472 break;
473
474
475 case filterType::highShelf1:
476
477 b_0 = one + ((minusOne + (a * a)) / (one + omega));
478 b_1 = minusOne * (((one - omega) / (one + omega)) + ((minusOne + (a * a)) / (one + omega)));
479 b_2 = zero;
480 a_0 = one;
481 a_1 = minusOne * ((one - omega) / (one + omega));
482 a_2 = zero;
483
484 break;
485
486
487 case filterType::highShelf1C:
488
489 b_0 = one + ((minusOne + (a * a)) / (one + (omega * a)));
490 b_1 = minusOne * (((one - (omega * a)) / (one + (omega * a))) + ((minusOne + (a * a)) / (one + (omega * a))));
491 b_2 = zero;
492 a_0 = one;
493 a_1 = minusOne * ((one - (omega * a)) / (one + (omega * a)));
494 a_2 = zero;
495
496 break;
497
498
499 case filterType::peak:
500
501 b_0 = one + (alpha * a);
502 b_1 = minusTwo * cos;
503 b_2 = one - (alpha * a);
504 a_0 = one + (alpha / a);
505 a_1 = minusTwo * cos;
506 a_2 = one - (alpha / a);
507
508 break;
509
510
511 case filterType::notch:
512
513 b_0 = one;
514 b_1 = minusTwo * cos;
515 b_2 = one;
516 a_0 = one + alpha;
517 a_1 = minusTwo * cos;
518 a_2 = one - alpha;
519
520 break;
521
522
523 case filterType::allPass:
524
525 b_0 = one - alpha;
526 b_1 = minusTwo * cos;
527 b_2 = one + alpha;
528 a_0 = one + alpha;
529 a_1 = minusTwo * cos;
530 a_2 = one - alpha;
531
532 break;
533
534
535 default:
536
537 b_0 = one;
538 b_1 = zero;
539 b_2 = zero;
540 a_0 = one;
541 a_1 = zero;
542 a_2 = zero;
543
544 break;
545 }
546
547 a0 = (one / a_0);
548 a1 = ((-a_1) * a0);
549 a2 = ((-a_2) * a0);
550 b0 = (b_0 * a0);
551 b1 = (b_1 * a0);
552 b2 = (b_2 * a0);
553}

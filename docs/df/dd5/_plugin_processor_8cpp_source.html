146void BiquadsAudioProcessor::prepareToPlay (double sampleRate, int samplesPerBlock)
147{
148 juce::ignoreUnused(sampleRate, samplesPerBlock);
149
150 processingPrecision = getProcessingPrecision();
151
152 spec.sampleRate = getSampleRate();
153 spec.maximumBlockSize = getBlockSize();
154 spec.numChannels = getTotalNumInputChannels();
155
156 rmsLeft.reset(sampleRate, rampDurationSeconds);
157 rmsRight.reset(sampleRate, rampDurationSeconds);
158
159 rmsLeft.setCurrentAndTargetValue(-100.0f);
160 rmsRight.setCurrentAndTargetValue(-100.0f);
161
162 processorFloat.prepare( getSpec() );
163 processorDouble.prepare( getSpec() );
164}

184void BiquadsAudioProcessor::processBlock(juce::AudioBuffer<float>& buffer, juce::MidiBuffer& midiMessages)
185{
186 if (bypassState->get())
187 {
188 processBlockBypassed(buffer, midiMessages);
189 }
190
191 else
192 {
193 juce::ScopedNoDenormals noDenormals;
194
195 processorFloat.process(buffer, midiMessages);
196
197 rmsLeft.skip(buffer.getNumSamples());
198 rmsRight.skip(buffer.getNumSamples());
199
200 {
201 const auto value = juce::Decibels::gainToDecibels(buffer.getRMSLevel(0, 0, buffer.getNumSamples()));
202 if (value < rmsLeft.getCurrentValue())
203 rmsLeft.setTargetValue(value);
204 else
205 rmsLeft.setCurrentAndTargetValue(value);
206 }
207
208 {
209 const auto value = juce::Decibels::gainToDecibels(buffer.getRMSLevel(1, 0, buffer.getNumSamples()));
210 if (value < rmsRight.getCurrentValue())
211 rmsRight.setTargetValue(value);
212 else
213 rmsRight.setCurrentAndTargetValue(value);
214 }
215 }
216}

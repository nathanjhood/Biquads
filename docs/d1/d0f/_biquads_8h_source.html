100 template <typename ProcessContext>
101 void process(const ProcessContext& context) noexcept
102 {
103 const auto& inputBlock = context.getInputBlock();
104 auto& outputBlock = context.getOutputBlock();
105 const auto numChannels = outputBlock.getNumChannels();
106 const auto numSamples = outputBlock.getNumSamples();
107 const auto len = inputBlock.getNumSamples();
108
109 jassert(inputBlock.getNumChannels() == numChannels);
110 jassert(inputBlock.getNumSamples() == numSamples);
111
112 if (context.isBypassed)
113 {
114 frq.skip(static_cast<int> (len));
115 res.skip(static_cast<int> (len));
116 lev.skip(static_cast<int> (len));
117
118 outputBlock.copyFrom(inputBlock);
119 return;
120 }
121
122 for (size_t channel = 0; channel < numChannels; ++channel)
123 {
124 auto* inputSamples = inputBlock.getChannelPointer(channel);
125 auto* outputSamples = outputBlock.getChannelPointer(channel);
126
127 for (size_t i = 0; i < numSamples; ++i)
128 outputSamples[i] = processSample((int)channel, inputSamples[i]);
129 }
130
131#if JUCE_DSP_ENABLE_SNAP_TO_ZERO
132 snapToZero();
133#endif
134 }

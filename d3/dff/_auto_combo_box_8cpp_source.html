52AutoComboBox::AutoComboBox(juce::AudioProcessor& p, APVTS& apvts, std::function<void()> paramLambda)
53{
54 auto addBox = [=, &apvts](juce::AudioParameterChoice* param)
55 {
56 BoxWithAttachment* newBox = new BoxWithAttachment;
57
58 addAndMakeVisible(newBox->box);
59
60 newBox->box.setName(param->name);
61 newBox->box.addItemList(param->choices, 1);
62 newBox->box.setSelectedItemIndex(0);
63 newBox->box.onChange = paramLambda;
64
65 newBox->attachment.reset(new ComboBoxAttachment(apvts, param->paramID, newBox->box));
66
67 boxes.add(newBox);
68 };
69
70 auto params = p.getParameters();

72 for (auto* param : params)
73 {
74 //======================================================================

22Viewer::Viewer(juce::AudioProcessorValueTreeState& apvts, juce::dsp::ProcessSpec& spec) :
23 apvts(apvts), setup(spec),
24 forwardFFT(order)
25{
26 needsCurveUpdate = false;
27 startTimerHz(27);
28
29 eqFreqParameter = apvts.getRawParameterValue("frequencyID");
30 eqQParameter = apvts.getRawParameterValue("resonanceID");
31 eqGainParameter = apvts.getRawParameterValue("gainID");
32 transParameter = apvts.getRawParameterValue("transformID");
33 typeParameter = apvts.getRawParameterValue("typeID");
34
35 dryFFTBuffer.reset(new float[2 * size]);
36 wetFFTBuffer.reset(new float[2 * size]);
37 H.reset(new float[size]);
38
39 dryBuffer.setSize(1, size);
40
41 const float beta = (float)size * logf(fHigh / fLow);
42 const float phi = juce::MathConstants<float>::pi / 180.0f;
43
44 for (int n = 0; n < size; ++n)
45 {
46 float phase = juce::MathConstants<float>::twoPi * beta * fLow * (powf(fHigh / fLow, (float)n / (float)size) - 1.0f);
47 dryBuffer.setSample(0, n, cosf((phase + phi) / fs));
48 dryFFTBuffer[n] = dryBuffer.getSample(0, n);
49 }
50
51 forwardFFT.performFrequencyOnlyForwardTransform(dryFFTBuffer.get());
52
53 wetBuffer.setSize(1, size);
54
55 //processor.toggleOnOff(true);
56 processor.prepare(setup);
57 processBuffer();
58}

73void Viewer::processBuffer()
74{
75 processor.setFrequency(eqFreqParameter->load());
76 processor.setResonance(eqQParameter->load());
77 processor.setGain(juce::Decibels::decibelsToGain(eqGainParameter->load()));
78 processor.setFilterType(static_cast<FilterType> ((int)typeParameter->load()));
79 processor.setTransformType(static_cast<TransformationType>((int)transParameter->load()));
80
81 wetBuffer.clear();
82 wetBuffer.makeCopyOf(dryBuffer);
83
84 processor.reset((double)fs);
85
86 juce::dsp::AudioBlock<float> block(wetBuffer);
87 juce::dsp::ProcessContextReplacing context(block);
88
89 processor.process(context);
90
91 for (int n = 0; n < size; ++n)
92 {
93 wetFFTBuffer[n] = wetBuffer.getSample(0, n);
94 }
95
96 forwardFFT.performFrequencyOnlyForwardTransform(wetFFTBuffer.get());
97
98 float HTemp[size];
99 for (int n = 0; n < size; ++n)
100 {
101 HTemp[n] = std::abs(wetFFTBuffer[n] / dryFFTBuffer[n]);
102 }
103
104 const int avgLength = 50;
105 for (int n = 0; n < size - avgLength; ++n)
106 {
107 float sum = 0.0f;
108 for (int k = 0; k < avgLength; ++k)
109 sum += HTemp[n + k];
110 H[n] = sum / (float)avgLength;
111 }
112}

114float Viewer::getMagnitudeForX(float x)
115{
116 float width = (float)getWidth();
117 float freq = fLow * pow((fHigh / fLow), ((x - 2.5f) / (width - 5.0f)));
118 int index = int((size / 2) * freq / (fs / 2.0f));
119
120 return H[index];
121}

123void Viewer::updateCurve()
124{
125 processBuffer();
126
127 curvePath.clear();
128 bool started = false;
129 const float scaleFactor = ((getHeight() / 2) - (getHeight()) / (8)) / 15.0f;
130 for (float xPos = 0.0f; xPos < (float)getWidth(); xPos += 1.0f)
131 {
132 auto traceMag = juce::Decibels::gainToDecibels(getMagnitudeForX(xPos));
133 auto traceY = (getHeight() / 2) - (traceMag * scaleFactor);
134
135 if (!started)
136 {
137 curvePath.startNewSubPath(xPos, traceY);
138 started = true;
139 }
140 else
141 {
142 curvePath.lineTo(xPos, traceY);
143 }
144 }
145
146 repaint();
147}
